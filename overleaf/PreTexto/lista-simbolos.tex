%%%% LISTA DE SÍMBOLOS
%%
%% Conjunto de sinais que substituem o nome de uma coisa ou de uma ação.
%%
%% Este arquivo para definição de símbolos é utilizado com a opção "file" (arquivo)

\listadesimbolos%% Formatação da lista de símbolos

\begin{listadeletraslatinas}%% Ambiente listadeletraslatinas (Letras latinas: Símbolo & Definição & \unidade{Unidade} \\)
$x$ & Valor observado ou variável aleatória & \\
$\bar{x}$ & Média amostral & \\
$s$ & Desvio-padrão amostral corrigido & \\
$n$ & Tamanho da amostra & \\
$k$ & Número de clusters & \\
$J$ & Função objetivo do K-Means & \\
$h(x)$ & Comprimento do caminho no Isolation Forest & \\
$c(\psi)$ & Fator de normalização do Isolation Forest & \\
$\psi$ & Número de observações na subamostra & \\
$\Delta$ & Ganho estimado (diferença) & \\
$\Delta_{\%}$ & Ganho percentual & \\
\end{listadeletraslatinas}

\begin{listadeletrasgregas}%% Ambiente listadeletrasgregas (Letras gregas: Símbolo & Definição & \unidade{Unidade} \\)
$\mu$ & Média populacional ou média do cluster & \\
$\sigma$ & Desvio-padrão populacional ou desvio-padrão do cluster & \\
\end{listadeletrasgregas}

\begin{listadenotacoes}%% Ambiente listadenotacoes (Notações: Símbolo & Definição & \unidade{Unidade} \\)
$z$ & Z-score (medida de desvio padrão) & \\
$p_{50}$ & Percentil 50 (mediana) & \\
$p_{75}$ & Percentil 75 & \\
$p_{90}$ & Percentil 90 & \\
$p_{95}$ & Percentil 95 & \\
$p_{99}$ & Percentil 99 & \\
$s(x, \psi)$ & Score de anomalia do Isolation Forest & \\
$E(h(x))$ & Valor esperado do comprimento do caminho & \\
$w_{ij}$ & Indicador de pertencimento ao cluster (1 se $x_i$ pertence ao cluster $j$, 0 caso contrário) & \\
$\|x_i - \mu_j\|^2$ & Distância euclidiana ao quadrado entre observação e centróide & \\
$d(x, y)$ & Distância euclidiana entre dois pontos & \\
\end{listadenotacoes}
